pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                sh """
					cd myproject
					mvn clean compile
				"""
            }
        }
        stage('Test') {
            steps {
                sh """
					cd myproject
					mvn test
				"""
            }
        }
        stage('Deploy') {
            steps {
                sh"""
					cd myproject
					mvn package
				"""
            }
        }
    }
}
